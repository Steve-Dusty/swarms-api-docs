**Swarm Type**: `HeavySwarm`

## Overview

The HeavySwarm is designed for complex, resource-intensive tasks that require deep analysis, multiple iterations, and extensive processing. It employs a question-answer pattern with specialized worker agents that can tackle challenging problems through iterative refinement and comprehensive exploration. This architecture is ideal for tasks that would overwhelm simpler swarm configurations.

Key features:
- **High-Capacity Processing**: Handles complex, multi-faceted tasks
- **Iterative Refinement**: Multiple loops per agent for deep analysis
- **Question-Answer Pattern**: Question agent guides worker exploration
- **Configurable Models**: Separate models for questioning and working
- **Extended Processing**: Designed for longer-running, thorough tasks

## Use Cases

- Complex research and analysis requiring multiple perspectives
- Deep technical investigations
- Comprehensive due diligence processes
- Multi-stage document analysis
- Strategic planning with thorough exploration
- Scientific hypothesis evaluation

import requests
import json

API_BASE_URL = "https://api.swarms.world"
API_KEY = "your_api_key_here"

    "x-api-key": API_KEY,
    "Content-Type": "application/json"

    "name": "Comprehensive Due Diligence",
    "description": "Heavy swarm for thorough company analysis",
    "swarm_type": "HeavySwarm",

    risks, and provide acquisition recommendation.""",

    "heavy_swarm_question_agent_model_name": "gpt-4.1",
    "heavy_swarm_worker_model_name": "claude-sonnet-4-20250514",
    "max_loops": 1
}

response = requests.post(
    f"{API_BASE_URL}/v1/swarm/completions",
    headers=headers,
    json=swarm_config,
    timeout=600  # Extended timeout for heavy processing

if response.status_code == 200:
    result = response.json()
    print("HeavySwarm analysis completed!")
    print(f"Cost: ${result['metadata']['billing_info']['total_cost']}")
    print(f"Execution time: {result['metadata']['execution_time_seconds']} seconds")
    print(f"\n=== Analysis Results ===")
    print(result['output'])
else:
    print(f"Error: {response.status_code} - {response.text}")

package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"

    "time"

type SwarmConfig struct {
    Name                            string `json:"name"`
    Description                     string `json:"description"`
    SwarmType                       string `json:"swarm_type"`
    Task                            string `json:"task"`
    HeavySwarmLoopsPerAgent         int    `json:"heavy_swarm_loops_per_agent"`
    HeavySwarmQuestionAgentModelName string `json:"heavy_swarm_question_agent_model_name"`
    HeavySwarmWorkerModelName       string `json:"heavy_swarm_worker_model_name"`
    MaxLoops                        int    `json:"max_loops"`
}

func main() {
    API_BASE_URL := "https://api.swarms.world"
    API_KEY := "your_api_key_here"

    swarmConfig := SwarmConfig{
        Name:                            "Comprehensive Due Diligence",
        Description:                     "Heavy swarm for thorough company analysis",
        SwarmType:                       "HeavySwarm",
        Task:                            "Conduct comprehensive due diligence on TechStartup Inc...",
        HeavySwarmLoopsPerAgent:         3,
        HeavySwarmQuestionAgentModelName: "gpt-4.1",
        HeavySwarmWorkerModelName:       "claude-sonnet-4-20250514",
        MaxLoops:                        1,
    }

    jsonData, _ := json.Marshal(swarmConfig)

    client := &http.Client{Timeout: 600 * time.Second}
    req, _ := http.NewRequest("POST", API_BASE_URL+"/v1/swarm/completions", bytes.NewBuffer(jsonData))
    req.Header.Set("x-api-key", API_KEY)
    req.Header.Set("Content-Type", "application/json")

    resp, err := client.Do(req)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Printf("Response: %s\n", string(body))
}

## Configuration Parameters

| Parameter | Type | Default | Description |
|-----------|------|---------|-------------|
| `heavy_swarm_loops_per_agent` | integer | 1 | Number of iterations each agent performs |
| `heavy_swarm_question_agent_model_name` | string | "gpt-4.1" | Model for the question/orchestration agent |
| `heavy_swarm_worker_model_name` | string | "claude-sonnet-4-20250514" | Model for worker agents |
| `max_loops` | integer | 1 | Overall swarm loop iterations |

## Best Practices

- Use HeavySwarm for tasks requiring thorough, multi-faceted analysis
- Set `heavy_swarm_loops_per_agent` to 2-5 for comprehensive exploration
- Use more capable models for complex reasoning tasks
- Increase timeout values as HeavySwarm processing takes longer
- Structure tasks with clear analytical dimensions for better coverage
- Consider cost implications - HeavySwarm uses more tokens than lighter swarms
- Use streaming for better UX during long-running analyses
